async function request_news_author(url) {
    //console.log(`received url: ${url}`)
    return new Promise((resolve, reject) => {
        request(url, function (error, response, body) {
            if (!error) {
                let $ = cheerio.load(body);
                let data = $("#Content1");
                let author = data.find('p').html().split('／')[1];
                //console.log(author);
                resolve(author);
            } else {
                reject(error);
            }

    });
